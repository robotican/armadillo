cmake_minimum_required(VERSION 2.8.3)
project(armadillo_services)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED COMPONENTS
        armadillo_msgs
        roscpp
#        std_msgs
        message_generation
        genmsg
)

catkin_package(
  INCLUDE_DIRS include
#  LIBRARIES armadillo2_msgs

        message_runtime
        roscpp
        armadillo_msgs
#  DEPENDS system_lib
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}

        src/dxl_torque.cpp
        src/lift_arm.cpp
        src/pan_tilt_mover.cpp
        src/shutdown.cpp
        include/armadillo_services/dxl_torque.h
        include/armadillo_services/lift_arm.h
        include/armadillo_services/pan_tilt_mover.h

## Specify libraries to link a library or executable target against
target_link_libraries(${PROJECT_NAME}_node
   ${catkin_LIBRARIES}
